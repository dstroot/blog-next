OpenAI's Codex was trained on publicly available source code and natural language, so it understands **both programming and human languages**. const debounce = (callback, wait) => {
    let timeoutId = null;

    return (...args) => {
        window.clearTimeout(timeoutId);

        timeoutId = window.setTimeout(() => {
            callback.apply(null, args);
        }, wait);
    };

};
